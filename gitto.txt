def print_board(board):
    print("\nBoard Positions:")
    for i in range(3):
        print(" | ".join(board[i]))
        if i < 2:
            print("--+---+--")
    print()


def check_win(board):
    def val(x):
        return int(x) if x != " " else 0
    lines = []
    for row in board:
        lines.append([val(x) for x in row])
    for col in range(3):
        lines.append([val(board[row][col]) for row in range(3)])
    lines.append([val(board[i][i]) for i in range(3)])
    lines.append([val(board[i][2 - i]) for i in range(3)])

    # Check sum = 15
    for line in lines:
        if sum(line) == 15 and 0 not in line:
            return True

    return False


def is_draw(used_numbers):
    return len(used_numbers) == 9


def get_position():
    while True:
        try:
            pos = int(input("Enter position (1-9): "))
            if 1 <= pos <= 9:
                return pos
            else:
                print("Invalid position! Choose 1-9.")
        except:
            print("Invalid input! Enter a number.")


def get_number(used_numbers):
    while True:
        try:
            num = int(input("Enter number (1-9): "))
            if num < 1 or num > 9:
                print("Number must be between 1-9.")
            elif num in used_numbers:
                print("Number already used!")
            else:
                return num
        except:
            print("Invalid input! Enter a number.")


def main():
    board = [[" " for _ in range(3)] for _ in range(3)]
    used_numbers = set()
    current_player = 1

    print("Magic Square Game Start!")

    while True:
        print_board(board)
        print(f"Player {current_player}'s turn")

        pos = get_position()
        row = (pos - 1) // 3
        col = (pos - 1) % 3

        if board[row][col] != " ":
            print("Position already filled! Try again.")
            continue

        num = get_number(used_numbers)

        board[row][col] = str(num)
        used_numbers.add(num)
        if check_win(board):
            print_board(board)
            print(f"Player {current_player} wins!")
            break
        if is_draw(used_numbers):
            print_board(board)
            print("It's a draw!")
            break
        current_player = 2 if current_player == 1 else 1

main()
